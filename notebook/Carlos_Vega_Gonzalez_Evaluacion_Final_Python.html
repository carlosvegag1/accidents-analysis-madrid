def generar_diccionario_mapeo(columna):

    """
    Genera un diccionario de mapeo para reducir categoras en una columna dada.

    Parmetros:
    columna (pd.Series): Columna del DataFrame que contiene las categoras a reducir.

    Retorna:
    dict: Diccionario con las categoras originales como claves y las categoras reducidas como valores.
    """
    
    categorias_unicas = columna.unique()
    print("Primero, hay que definir las categoras en las que se va a reducir esta columna:")
    categorias = []
    numero_categorias = int(input("Cuntas categoras principales deseas definir? "))
    for i in range(numero_categorias):
        cat = input(f"Introduce la categora principal {i + 1}: ")
        categorias.append(cat)

    mapeo = {}
    print("\nAhora, vamos a asignar las categoras:")
    for categoria in categorias_unicas:
        print(f"\nTienes que reestructurar la categora: {categoria}")
        print("Recuerda, debes escoger entre estas:")
        for i, principal in enumerate(categorias, start=1):
            print(f"{i} = {principal}")
        while True:
            try:
                opcion = int(input(f"A qu categora se asignar '{categoria}'? (Teclea el nmero): "))
                if 1 <= opcion <= len(categorias):
                    mapeo[categoria] = categorias[opcion - 1]
                    break
                else:
                    print("Opcin fuera de rango. Intntalo de nuevo.")
            except ValueError:
                print("Entrada no vlida. Teclea un nmero.")

    return mapeo
